Процедура для возврата по дереву на InterBase;
CREATE PROCEDURE prnt (ID INTEGER)
RETURNS (DID INTEGER, OID INTEGER, NAME VARCHAR(30))
AS
BEGIN
  WHILE (:ID > 0) DO
    BEGIN
      SELECT O.ID, O.PRNT, O.SOMETHING
      FROM OBJECTS O
      WHERE O.ID = :ID
      INTO :DID, :OID, :NAME;
      ID = :OID; 
      SUSPEND;
    END
END
